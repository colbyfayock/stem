<script id="welcome-message" type="text/x-handlebars-template">
    <h1>Hi there, I'm {{ name }}</h1>
    <h2>Some cool things to do are:</h2>
    {{> list}}
</script>

<script id="list" type="text/x-handlebars-template">
    {{# if things }}

            {{#things}}
                <li>{{name}}</li>
            {{/things}}
        </ul>
    {{/ if }}
</script>
